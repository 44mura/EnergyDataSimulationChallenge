days = pd.Series('Mon Tue Wed Thu Fri Sat Sun'.split())
df_energy_per_day = df_total_watt.resample('d', how='mean')
kmeans = KMeans(n_clusters=3)
kmeans.fit([[x] for x in df_energy_per_day.Energy.fillna(0).tolist()])
df_energy_per_day['label']=kmeans.labels_    

# For sorting label (;;)
depd = df_energy_per_day
label_sorted = sorted([depd.ix[depd.label == i].ix[0][0] for i in range(3)])
color_sorted = [0]*3
for i, label in enumerate(map(lambda x: depd.ix[depd.Energy == x].ix[0][1], label_sorted)):
    color_sorted[int(label)]=colors[i]

plt.figure(figsize=(16, 6))
for day in df_energy_per_day.groupby(lambda x: x.weekday):
    for row in range(day[1].Energy.count()):
        color = color_sorted[int(day[1].iloc[[row]].label)]
        plt.bar(0.1+ day[0] + float(row)/7, day[1].iloc[[row]].Energy, 0.12, color=color)

plt.xticks(np.arange(0.5,7.5,1), days)
for x in [1.03,2.01,2.95,3.95,4.95,5.95]:
    plt.axvline(x=x, linestyle='--', color='grey')
plt.xlabel('day of the week')
plt.ylabel('Energy Consumption in a day')
plt.title('Difference of Energy Consumption per day of the week')

p = [0]*3
for i in range(3):
    p[i] = plt.Rectangle((0, 0), 1, 1, fc=colors[i])
plt.legend([p[2],p[1],p[0]], ["high","medium","low"], loc=2)
plt.show()

In this chart, the data is clustered in three groups (high, medium and low) using the kmeans algorithm, and the result is showen with a bar chart per day of the week. The chart shows the weekly variation of energy consumption per day of the week in a quantitative way.
